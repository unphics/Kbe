环境变量:
    KBE_ROOT: 你的目录\kbengine
    KBE_BIN_PATH: %KBE_ROOT%/kbe/bin/server/
    KBE_RES_PATH: %KBE_ROOT%/kbe/res;%KBE_ROOT%/assets/;%KBE_ROOT%/assets/srcipt;%KBE_ROOT%/assets/res/

mysql:
    数据库名称: kbe
    用户名: kbeuser
    密码: 123456

mysql创建新账户与赋权操作:
    create user kbeuser@localhost identified by '123456';
//  grant all privileges on *.* to kbeuser@localhost with grant option;

mysql为course_asset操作:
    create database kbecourse;
    use mysql;
    delete from user where user='';
    flush privileges;
    grant all privileges on kbecourse.* to kbeuser@localhost with grant option;
    grant select,insert,update,delete,create,drop on kbecourse.* to kbeuser@localhost with grant option;
    flush privileges;

mysql常见指令:
    查看所有账户: SELECT user, host FROM mysql.user;
    删除用户: DROP USER 'username'@'host';
    查看database列表: SHOW DATABASES;
    删除一个database: DROP DATABASE dbname;

mysql ssl 问题:
    ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY '123456';
    flush privileges;
    将mysql data目录的my.ini下[mysqld]后面增加
        # disable_ssl
        skip_ssl
        ssl=0
    重启mysql服务
    ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY '123456';
    flush privileges;
