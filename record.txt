    KBE_ROOT: 你的目录\kbengine
    KBE_BIN_PATH: %KBE_ROOT%/kbe/bin/server/
    KBE_RES_PATH: %KBE_ROOT%/kbe/res;%KBE_ROOT%/assets/;%KBE_ROOT%/assets/srcipt;%KBE_ROOT%/assets/res/

mysql为course_asset操作:
    create database kbecourse;
    use mysql;
    delete from user where user='';
    flush privileges;
    grant all privileges on kbecourse.* to kbeuser@localhost with grant option;
    grant select,insert,update,delete,create,drop on kbecourse.* to kbeuser@localhost with grant option;
    flush privileges;

    flush privileges;
    将mysql data目录的my.ini下增加
        # disable_ssl
        skip-ssl
        ssl=0
